1176  JSON parse_string( const string &str, size_t &offset, bool &ok ) {
1177  string val;
1178  for( char c = str[++offset]; c != '\"' ; c = str[++offset] ) {
1179  if( c == '\\' ) {
1180  switch( str[ ++offset ] ) {
1181  case '\"': val += '\"'; break;
1182  case '\\': val += '\\'; break;
1183  case '/' : val += '/' ; break;
1184  case 'b' : val += '\b'; break;
1185  case 'f' : val += '\f'; break;
1186  case 'n' : val += '\n'; break;
1187  case 'r' : val += '\r'; break;
1188  case 't' : val += '\t'; break;
1189  case 'u' : {
1190  val += "\\u" ;
1191  for( unsigned i = 1; i <= 4; ++i ) {
1192  c = str[offset+i];
1193  if( (c >= '0' && c <= '9') || (c >= 'a' && c <= 'f') || (c >= 'A' && c <= 'F') )
1194  val += c;
1195  else {
1196  std::cerr << "Error: String: Expected hex character in unicode escape, found '" << c << "'";
1197  ok = false;
1198  return JSON::Make( JSON::Class::String );
1199  }
1200  }
1201  offset += 4;
1202  } break;
1203  default : val += '\\'; break;
1204  }
1205  }
1206  else
1207  val += c;
1208  }
1209  ++offset;
1210  return JSON(val);
1211  }
